class Ball
{
	int x;
	int y;
	public Ball(int xPos, int yPos) // constructor
	{
		this.x = xPos;
		this.y = yPos;
	}
	public Ball()
	{
		
	}
	public void printInfo()
	{
		System.out.println("I am a ball at " + x + ":" + y);
		System.out.println("My address = " + this);
	}
}

	}
