class Shape
{
	private int xPos;
	private int yPos;	
	public void printPositions()
	{
		System.out.println("Shape at pos " + xPos + ":" + yPos);		
	}
	public void draw() 
	{
		System.out.println("I am some shape");
	}
}
class Circle extends Shape
{
	public void draw() {  // overriding function from base (parent) class
		System.out.println("O");
		printPositions();
	}
}
class Rectangle extends Shape
{
	public void draw() {   // overriding function from base class
		System.out.println("[]");
		printPositions();
	}	
}
public class Example {

	public static void main(String[] argsadfadf) {
		Shape shape = new Rectangle();  
		shape.draw();
		
		shape = new Circle();
		shape.draw();

	}
